  --output_file results/annot/sample_vep.txt \
  --offline --everything --force_overwrite

  --input results/mutect/sample.vcf.gz \
  --type vcf \
  --annot results/annot/sample_vep.txt \
  --out results/annot/sample_annotated.tsv
